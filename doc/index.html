C IMPLICIT NONE INCLUDE 'PARAM_UMAT.INC' C INTEGER IP,IQ,N,NMAX,NP,NROT,I,J PARAMETER (NMAX=100) C DOUBLE PRECISION A(NP,NP),D(NP),V(NP,NP),B(NMAX),Z(NMAX), + SM,TRESH,G,T,H,THETA,S,C,TAU
